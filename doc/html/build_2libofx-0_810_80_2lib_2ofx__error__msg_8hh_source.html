130 const ErrorMsg find_error_msg(int param_code)
131 {
132  ErrorMsg return_val;
133  int i;
134  bool code_found = false;
135 
136  for (i = 0; i < 2000 && (code_found == false); i++)
137  {
138  if ((error_msgs_list[i].code == param_code) || (error_msgs_list[i].code == -1))
139  {
140  return_val = error_msgs_list[i];
141  code_found = true;
142  }
143  }
144  return return_val;
145 }
